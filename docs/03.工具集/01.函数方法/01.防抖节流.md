### 防抖
```javascript
/**
 * @description: 防抖
 * @param {*} delay
 * @param {*} handle
 * @return {*}
 */
export function debounce(handle, delay) {
  var timer = null;
  return function () {
    var _self = this,
      _args = arguments;
    clearTimeout(timer);
    timer = setTimeout(function () {
      handle.apply(_self, _args)
    }, delay)
  }
}
```
### 节流
```javascript
/**
 * @description: 节流
 * @param {*} interval
 * @param {*} fn
 * @return {*}
 */
export function throttle(fn, interval) {
  var _self = fn, // 保存需要被延迟执行的函数引用
    timer, // 定时器
    firstTime = false; // 是否是第一次调用
  return function () {
    var args = arguments;
    var _me = this;
    if (firstTime) { // 如果是第一次调用不需要延迟
      _self.apply(_me, args); // 执行fn函数并且修正此函数中this所运行的上下文指向
      return firstTime = false;
    }
    if (timer) { // 如果定时器还在，说明前一次延迟执行还没有完成
      return false;
    }
    timer = setTimeout(function () { // 延迟一段时间执行
      clearTimeout(timer);
      timer = null;
      _self.apply(_me, args); // 执行fn函数并且修正此函数中this所运行的上下文指向
    }, interval || 500);
  }
}
```

### Vue中使用步骤
```vue
<script>
import { throttle, debounce } from "@/util/util";
export default {
  methods: {
    getListThrottle: throttle(function () {
      this.getList();
    }, 1000),
    getList() {
      // 请求数据
    }
  }
}
</script>
<!--以上是防抖的示例，节流同理-->
```
